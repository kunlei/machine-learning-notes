8.1 Decision Tree

8.1.1 Classification

from sklearn.datasets import load_breast_cancer
from sklearn.tree import DecisionTreeClassifier
from sklearn.model_selection import train_test_split

cancer = load_breast_cancer()
X_train, X_test, y_train, y_test = train_test_split(cancer['data'], cancer['target'], random_state=42)
clf = DecisionTreeClassifier()
clf.fit(X_train, y_train)
print(f'train score: {clf.score(X_train, y_train):.2f}')
print(f'test score: {clf.score(X_test, y_test):.2f}')
train score: 1.00
test score: 0.94

clf.fit(X_train, y_train)

8.1.3 Regression

from sklearn.tree import DecisionTreeRegressor
from sklearn.datasets import load_diabetes
from sklearn.model_selection import train_test_split

diabetes = load_diabetes()
X_train, X_test, y_train, y_test = train_test_split(diabetes['data'], diabetes['target'], random_state=42)
tree = DecisionTreeRegressor(max_depth=5)
tree.fit(X_train, y_train)
print(f'train score: {tree.score(X_train, y_train):.2f}')
print(f'test score: {tree.score(X_test, y_test):.2f}')
train score: 0.69
test score: 0.23
from sklearn.tree import export_graphviz
import graphviz

export_graphviz(tree, out_file="tree.dot",
feature_names=diabetes.feature_names, impurity=False, filled=True)
with open("tree.dot") as f: 
    dot_graph = f.read()

graphviz.Source(dot_graph)
import shap

explainer = shap.Explainer(tree.predict, X_train)
shap_values = explainer(X_test)
shap.plots.waterfall(shap_values[0])

8.3 Gradient Boosting Machines

8.3.1 Classification

from sklearn.datasets import load_breast_cancer
from sklearn.model_selection import train_test_split
from sklearn.ensemble import GradientBoostingClassifier

X, y = load_breast_cancer(return_X_y=True)
X_train, X_test, y_train, y_test = train_test_split(X, y, random_state=42)
clf = GradientBoostingClassifier(n_estimators=100, learning_rate=0.1, max_depth=3, random_state=42)
clf.fit(X_train, y_train)
print(f'train score: {clf.score(X_train, y_train):.2f}')
print(f'test score: {clf.score(X_test, y_test):.2f}')
train score: 1.00
test score: 0.96
